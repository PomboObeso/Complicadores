Funcao Inteiro fibonacci (Inteiro n) Inicio
	Se(n < 2) Inicio
		Devolve n ; 
	Fim
	Porem Inicio
		Devolve fibonacci(n - 1) + fibonacci(n - 2);
	Fim
Fim
